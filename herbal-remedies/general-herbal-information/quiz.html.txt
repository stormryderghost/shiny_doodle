<script>
function showResults() {
    var healthConcern = document.querySelector('input[name="health_concern"]:checked').value;

    switch(healthConcern) {
        case 'digestive':
            window.location.href = '/quiz-results-digestive.html';
            break;
        case 'anxiety':
            window.location.href = '/quiz-results-anxiety.html';
            break;
        case 'sleep':
            window.location.href = '/quiz-results-sleep.html';
            break;
        case 'immune':
            window.location.href = '/quiz-results-immune.html';
            break;
        case 'skin':
            window.location.href = '/quiz-results-skin.html';
            break;
        case 'pain':
            window.location.href = '/quiz-results-pain.html';
            break;
        case 'energy':
            window.location.href = '/quiz-results-energy.html';
            break;
        case 'respiratory':
            window.location.href = '/quiz-results-respiratory.html';
            break;
        case 'women':
            window.location.href = '/quiz-results-women.html';
            break;
        default:
            return false;
    }
    return false;
}
</script>
